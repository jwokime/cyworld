<script>
document.getElementById('game-poster-ramen').onclick = function() {
  this.style.display = 'none';
  var box = document.getElementById('game-box-ramen');
  box.style.display = 'block';
  setTimeout(function() {
    var ruffle = window.RufflePlayer.newest();
    var player = ruffle.createPlayer();
    player.style.width = '100%';
    player.style.height = '100%';
    box.appendChild(player);
    player.load({
      url: "https://jwokime.github.io/cyworld/scrapbook/games/sue-ramen.swf"
    });
  }, 500);
};
</script>
